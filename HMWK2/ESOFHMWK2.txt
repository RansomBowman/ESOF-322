package esof.hmwk.pkg2;

import java.io.IOException;
import java.util.Scanner;

/**
 *
 * @author ranso
 */
public class ESOFHMWK2 {

    /**
     * @param args the command line arguments
     */
        
        
    public static void main(String[] args) throws IOException{
        Scanner in = new Scanner(System.in);
        System.out.print("Input desired storage option (Relational, NoSQL, Graph): ");
        String input = in.next();
        input = input.toLowerCase();
        
        System.out.println("Input lowercase: " + input);
        
        if(input.equals("relational")){
            Relational rel = new Relational();
            rel.store();
        }else if(input.equals("nosql")){
            NoSQL nos = new NoSQL();
            nos.store();
        }else if(input.equals("graph")){
            Graph gra = new Graph();
            gra.store();
        }else{
            System.out.print("Invalid Input");
        }
        
      

    }
    
}
